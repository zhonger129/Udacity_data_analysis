# I need to make a function to convert the passegers' age and port of embarkation to work with my calcations

# < 10    : less_than_ten
# 10 - 20 : ten_to_twenty
# 20 - 30 : twenty_to_thirty
# 30 - 40 : thirty_to_fourty
# 40 - 50 : fourty_to_fifty
# 50 - 60 : fifty_to_sixty
# 60 - 70 : sixty_to_seventy
# > 70    : greater_than_seventy

def add_passenger_Age_stage(age):
    if age < 10:
        return 'less_than_ten'
    elif age < 20:
        return 'ten_to_twenty'
    elif age < 30:
        return 'twenty_to_thirty'
    elif age < 40:
        return 'thirty_to_fourty'
    elif age < 50:
        return 'fourty_to_fifty'
    elif age < 60:
        return 'fifty_to_sixty'
    elif age < 70:
        return 'sixty_to_seventy'
    elif age < 80:
        return 'greater_than_seventy'
    else:
        return None

# Add new columns to existing dataframe
def patch_df(titanic_df):
    titanic_df['Embarked_num'] = titanic_df['Embarked'].replace(['C','S','Q'],[1,2,3],inplace=False)
    titanic_df['Age_stage'] = titanic_df['Age'].apply(add_passenger_Age_stage)

# new dataframe to store passenger survived 
def survived_passenger_df(titanic_df):
    return titanic_df[titanic_df.Survived==1]

# Calculate the number of group data
def count_group_data(titanic_df,group_key):
    return titanic_df.groupby(group_key)['PassengerId'].count()

# Calculate the rate of survived passenger
def rate_of_survived_passenger(group_data,group_key):
    # Calculate number of all passenger in the group 
    all_passenger_of_each_group = count_group_data(titanic_df,group_key)

    # Calculate number of all passenger survived in the group 
    survived_passenger_of_each_group = count_group_data(survived_passenger_df(group_data),group_key)
    
    # Return the rate of survived passenger
    return survived_passenger_of_each_group / all_passenger_of_each_group

# Print the histogram chart 
def print_bar(group_data,title,xlab,ylab):
    sns.set_style("whitegrid")
    group_data.plot.bar(title=title,
                        figsize=(5,5),
                        color='b',
                        alpha=0.6
                       )
    xlabel(xlab,fontsize=12 )
    ylabel(ylab,fontsize=12 )
    

# Print the pie chart 
def print_pie(group_data,titel,label):
    figure
    figure(figsize=(6, 6))
    plt.pie(group_data,
            autopct='%3.1f%%',
            shadow=True,
            #explode=[0.1,0,0],
            startangle = 90
            #labels=label
           )
    plt.title(titel)
    legend(label)

Pearson correlation r¶

In [22]:
# This function is to calculate Pearson's r
# r = average of (x in std unit) * (y in std unit)
def correlation(x,y):
    std_x = (x - x.mean()) / x.std(ddof=0)
    std_y = (y - y.mean()) / y.std(ddof=0)
    
    return (std_x * std_y).mean()
